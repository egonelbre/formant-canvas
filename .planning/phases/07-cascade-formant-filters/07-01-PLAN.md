    ```typescript
    export function cascadeEnvelope(
      freq: number,
      formants: FormantParams[],
      order: FilterOrder = 2,
    ): number {
      let product = 1;
      for (const f of formants) {
        // Shape-only: use gain=1 so cascade product gives relative response
        const mag = formantMagnitude(freq, { freq: f.freq, bw: f.bw, gain: 1 });
        if (order === 4) {
          product *= mag * mag;
        } else {
          product *= mag;
        }
      }
      return product;
    }
    ```

    Add `topologyAwareEnvelope` function:
    ```typescript
    export function topologyAwareEnvelope(
      freq: number,
      formants: FormantParams[],
      topology: FilterTopology,
      order: FilterOrder = 2,
    ): number {
      if (topology === 'cascade') {
        return cascadeEnvelope(freq, formants, order);
      }
      // Parallel: sum of individual responses
      let sum = 0;
      for (const f of formants) {
        let mag = formantMagnitude(freq, f);
        if (order === 4) {
          mag = mag * mag;
        }
        sum += mag;
      }
      return sum;
    }
    ```
