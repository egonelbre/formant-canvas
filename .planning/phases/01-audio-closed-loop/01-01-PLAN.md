**3. Create `src/lib/audio/dsp/rosenberg.ts`:**
Pure function implementing Rosenberg C glottal pulse. Two phases: rising half-sinusoid (0 to Tp), cosine fall (Tp to Tn), zero (Tn to 1.0). Tp = 0.4 * openQuotient.

```typescript
export function rosenbergSample(phase: number, openQuotient: number = 0.6): number {
  if (phase < 0 || phase >= 1) return 0;
  const Tn = openQuotient;
  const Tp = 0.4 * Tn;
  if (phase < Tp) {
    return 0.5 * (1 - Math.cos(Math.PI * phase / Tp));
  } else if (phase < Tn) {
    const closingPhase = (phase - Tp) / (Tn - Tp);
    return Math.cos(Math.PI * 0.5 * closingPhase);
  }
  return 0;
}
```
