```typescript
function onPointerDown(e: PointerEvent) {
  e.preventDefault();
  const pos = pointerToSvg(e);
  if (!pos) return;
  svgEl?.setPointerCapture(e.pointerId);

  if (pos.svgY < HARMONIC_REGION_HEIGHT) {
    dragMode = 'formant';
    dragFormantIndex = findNearestFormant(pos.svgX);
    // D-14: temporarily override strategy when dragging a locked formant
    if (voiceParams.strategyMode === 'locked') {
      voiceParams.strategyOverriding = true;
    }
    const freq = xToFreq(pos.svgX);
    voiceParams[FORMANT_FREQ_KEYS[dragFormantIndex]] = Math.round(freq);
  } else {
    dragMode = 'piano';
    const midi = midiFromSvgX(pos.svgX, pos.svgY);
    if (midi !== null) voiceParams.f0 = midiToHz(midi);
  }
}
```

CRITICAL: This effect must ONLY read f0, strategyId, strategyMode, strategyOverriding, and voicePreset. It must NOT read the formant frequencies it writes to, to avoid circular reactive updates.

```typescript
// Strategy locked-mode: auto-tune formants to maintain ratio (D-11, STRAT-03)
// Placed BEFORE syncParams $effect so targets are written before audio sync.
// Guard: only write if value changes by >0.1 Hz to avoid circular updates (Pitfall 1).
$effect(() => {
  const mode = voiceParams.strategyMode;
  const id = voiceParams.strategyId;
  const f0 = voiceParams.f0;
  const overriding = voiceParams.strategyOverriding;
  const preset = voiceParams.voicePreset;

  if (mode !== 'locked' || overriding) return;

  const result = computeTargets(id, f0, preset ?? 'baritone');
  const t = result.targets;

  // Write targets to voiceParams -- the existing syncParams $effect will forward to audio
  // D-11: setTargetAtTime in AudioBridge provides smooth ~50-100ms interpolation
  if (t.f1 !== null) voiceParams.f1Freq = t.f1;
  if (t.f2 !== null) voiceParams.f2Freq = t.f2;
  if (t.f3 !== null) voiceParams.f3Freq = t.f3;
  if (t.f4 !== null) voiceParams.f4Freq = t.f4;
  if (t.f5 !== null) voiceParams.f5Freq = t.f5;
});
```

Note on Pitfall 1 (circular effects): This effect reads `voiceParams.strategyMode`, `voiceParams.strategyId`, `voiceParams.f0`, `voiceParams.strategyOverriding`, and `voiceParams.voicePreset`. It writes to `voiceParams.f1Freq` etc. Since Svelte 5 tracks reads, and the effect does NOT read f1Freq/f2Freq/etc., writing to those won't re-trigger this effect. The syncParams effect reads all via `voiceParams.snapshot`, which is fine -- it just pushes to audio.
